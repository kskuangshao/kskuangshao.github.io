row format delimited fields terminated by "\t";

要求使用sql统计出每个用户累计访问次数

select --第三步：窗口内累加
  userid,
  visitdate,
  cnt1,
  sum(cnt1) over(partition by userid order by visitdate) as cnt2 --注意这里窗口分区
from
(select --第二步：分组求和
  userid,
  visitdate,
  sum(visitcount) as cnt1
from
(select --第一步：把月份提取出来
  userid,
  date_format(regexp_replace(visitdate,'/','-'),'yyyyMM') as   visitdate,
  visitcount
from action)t1 group by userid,visitdate)t2;

建表语句
create table visit(user_id string,shop string) row format delimited fields terminated by '\t';

select
  shop,
  count(*) as num
 from
 (select
  shop,
  user_ida
from visit group by shop,user_id) t1 
group by shop; --uv记得去重


select --第三步：拿到前三
  shop,
  user_id,
  cnt
  from
(select --第二层：给每个访客排序，从上往下一次减少
  shop,
  user_id,
  cnt,
  row_number() over(partition by shop order by cnt desc) as rn
  from
(select --第一层：得到每个访客在每个店铺的次数
  user_id,
  shop,
  count(*) as cnt
from visit group by user_id,shop)t1)t2 where rn <= 3;

题目

  1. 蚂蚁森林植物申领统计

问题：假设2017年1月1日开始记录低碳数据（user_low_carbon），假设2017年10月1日之前满足申领条件的用户都申领了一颗 p004-胡杨，剩余的能量全部用来领取“p002-沙柳”。统计在10月1日累计申领“p002-沙柳”排名前10的用户信息；以及他比后一名多领了几颗沙柳。
得到的统计结果如下表样式：

  1. 蚂蚁森林低碳用户排名分析

问题：查询user_low_carbon表中每日流水记录，条件为：

  • 用户在2017年，连续三天（或以上）的天数里，
  • 每天减少碳排放（low_carbon）都超过100g的用户低碳流水。

需要查询返回满足以上条件的user_low_carbon表中的记录流水。

例如用户u_002符合条件的记录如下，因为2017/1/2~2017/1/5连续四天的碳排放量之和都大于等于100g：

1.取大于2017/1/1 小于2017/10/1每个用户的总排放量
select
    user_id,
    sum(low_carbon) as sum_low_carbon
from user_low_carbon
group by user_id;

2.胡杨碳排放
select
    plant_id,
    low_carbon
from plant_carbon
where plant_id = 'p004';

3.沙柳碳排放
select
    plant_id,
    low_carbon
from plant_carbon
where plant_id = 'p002';

4.每个用户沙柳的个数
select
    user_id,
    floor((sum_low_carbon-t2.low_carbon)/t3.low_carbon) as plant_count
from 
(select
    user_id,
    sum(low_carbon) as sum_low_carbon
from user_low_carbon
group by user_id)t1,
(select
    plant_id,
    low_carbon
from plant_carbon
where plant_id = 'p004')t2,
(select
    plant_id,
    low_carbon
from plant_carbon
where plant_id = 'p002')t3;

5.上推一格做出一列相减
select
    user_id,
    plant_count,
    lead(plant_count,1,'9999-99-99') over(order by plant_count desc) as lead_plant_count
from 
(select
    user_id,
    floor((sum_low_carbon-t2.low_carbon)/t3.low_carbon) as plant_count
from 
(select
    user_id,
    sum(low_carbon) as sum_low_carbon
from user_low_carbon
group by user_id)t1,

    plant_id,
    low_carbon
from plant_carbon

    plant_id,
    low_carbon
from plant_carbon

6.沙柳的个数相减
select

    user_id,

from 
(select
    user_id,

from user_low_carbon

    plant_id,

第二种方式（改进版）

第一种方式如果数据量很大，那么效率极低，因为我们不需要把所有的行全拿出来做运算和排序，我们只需要拿出top10的行并对其操作，所以改进如下

1.取大于2017/1/1 小于2017/10/1每个用户的总排放量
select
    user_id,
    sum(low_carbon) as sum_low_carbon
from user_low_carbon
group by user_id
order by sum_low_carbon desc
limit 11;--一开始就限制11条，因为用lead上提，所以需要多一行数据用来提上去
